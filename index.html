Integração Google Sheets
Como configurar:
  1. Crie uma planilha no Google Sheets
  2. Vá em Extensões → Apps Script
  3. Cole o script abaixo e publique como Web App
  4. Cole a URL gerada no campo abaixo
Code.gs — Cole no Google Apps Script 
function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet();
  var data = JSON.parse(e.postData.contents);
  
  if (data.action === 'sync') {
    var osSheet = sheet.getSheetByName('OS') || sheet.insertSheet('OS');
    if (osSheet.getLastRow() === 0) {
      osSheet.appendRow(['Numero','Data','Cliente','CNPJ','Endereco',
        'Tecnico','Marca','Modelo','Serie','Horimetro',
        'Tipo','Defeito','Servico','Pecas','Status','Tempo']);
    }
    data.data.forEach(function(o) {
      osSheet.appendRow([o.numero,o.data,o.cliente,o.cnpj,o.endereco,
        o.tecnico,o.marca,o.modelo,o.serie,o.horimetro,
        o.tipo,o.defeito,o.servico,o.pecas,o.status,o.tempo]);
    });
  }
  
  if (data.action === 'saveCliente') {
    var cliSheet = sheet.getSheetByName('Clientes') || sheet.insertSheet('Clientes');
    if (cliSheet.getLastRow() === 0) {
      cliSheet.appendRow(['ID','Nome','CNPJ','Endereco','Contato',
        'Telefone','Email','Marca','Modelo','Combustivel','Serie']);
    }
    var c = data.cliente;
    cliSheet.appendRow([c.id,c.nome,c.cnpj,c.end,c.contato,
      c.tel,c.email,c.marca,c.modelo,c.comb,c.serie]);
  }
  
  return ContentService.createTextOutput(
    JSON.stringify({status:'ok'})
  ).setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  if (e.parameter.action === 'ping') {
    return ContentService.createTextOutput(
      JSON.stringify({status:'ok'})
    ).setMimeType(ContentService.MimeType.JSON);
  }
}
